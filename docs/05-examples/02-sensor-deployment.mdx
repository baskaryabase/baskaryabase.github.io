```
# otel-agent resources
otel-agent:
  resources:
    limits:
      cpu: 1500m
      memory: 3072Mi
    requests:
      cpu: 1000m
      memory: 2048Mi
  tolerations:
    - key: "app"
      operator: "Equal"
      value: "kubesense"
      effect: "NoSchedule"
  affinity:
    nodeAffinity:
      requiredDuringSchedulingIgnoredDuringExecution:
        nodeSelectorTerms:
        - matchExpressions:
          - key: app
            operator: In
            values:
            - kubesense
```
